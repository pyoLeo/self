				cnt_a++;
			}
			else

		}
		
		System.out.println("자음 : " + cnt_q + "개 모음 : " + cnt_a + "개");

9. 사용자로부터 키를 입력받아서 표준 체중을 계산한 후에 사용자의 체중과 비교하여 저체중인지, 표준인지, 과체중인지를 판단하는 프로그램을 작성하라. 표준 체중 계산식은 다음을 사용하라.
표준체중 = ( 키 - 100 ) * 0.9
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		int height;
		int weight;
		double avg_weight;
		
		Scanner sc = new Scanner(System.in);
		System.out.print("키 : ");
		height = sc.nextInt();
		avg_weight = (height - 100) * 0.9;
		
		System.out.print("몸무게 : ");
		weight = sc.nextInt();
		
		if(weight > avg_weight) 
			System.out.println("과체중");
		else if(weight == avg_weight)
			System.out.println("정상");
		else
			System.out.println("저체중");
		
	}

10. 2와 100 사이에 있는 모든 소수(prime number)를 찾는 프로그램을 작성하라. 주어진 정수 k를 2부터 k-1까지의 숫자로 나누어서 나머지가 0인 것이 하나라도 있으면 소수가 아니다.
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		int cnt = 0;
		
		System.out.println("2부터 100 중 소수");
		for(int i = 2; i <= 100; i++) {
			for(int j = 2; j < i; j++) {
				if(i % j == 0) {
					cnt++;
				}
			}
			if(cnt == 0) 
				System.out.println(i);
			cnt = 0;
		}
	}
